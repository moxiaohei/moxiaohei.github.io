图像处理之浮雕效果的实现

1、MATLAB对数字图像的处理及浮雕效果的显示(以下代码亲测有效,小编用的是MATLABR2012b)
浮雕原理：实现图像浮雕效果的一般原理是，将图像上每个像素点与其对角线的像素点形成差值，使相似颜色值淡化，不同颜色值突出，从而产生纵深感，达到浮雕的效果，具体的做法是取主对角线除右下角外的各点之和的平均值，减去右下角点的值，再加上填充背景色，再加上一个背景常数，一般为128而成，会形成类似浮雕的效果
%算法1 直接对角线上左上减去右下加上一个常数
% y(i,j) = img(x-1,y-1)-img(x+1,y+1)+128
temp_img=imread(‘lena.jpg’); %需要改到你的本机的相应图片的路径下
size_info=size(temp_img);
height=size_info(1);
width=size_info(2);
spec_img=zeros(height,width,3);

for i=2:height-1
for j=2:width-1
spec_img(i,j,:)=double(temp_img(i-1,j-1,:))-double(temp_img(i+1,j+1,:))+128;
end
end
figure,imshow(temp_img);
figure,imshow(spec_img/255);

%算法2 取主对角线除右下角外的各点之和的平均值，减去右下角点的值，再加上填充背景色，
%再加上一个背景常数，会形成类似浮雕的效果。
%算法 y(i,j)= 0.25img(x-2,y-2)+0.25img(x-1,y-1)+0.25img(x,y)+0.25img(x+1,y+1)+0.25*img(x+2,y+1)+128;

A = imread(‘E:\MatlabProject\Project\1.jpg’);%需要改到你的本机的相应图片的路径下
size_info=size(A);
height=size_info(1);
width=size_info(2);
B=zeros(height,width,3);
% img_temp=rgb2gray(temp_img);

for i=2:height-1
for j=2:width-1
if A(i,j,:) == 255
A(i,j,:) = 255;
else
B(i,j,:)=double(A(i-1,j-1,:))-double(A(i+1,j+1,:))+128;
end
end
end
figure,imshow(A);
figure,imshow(B/255);

#include <graphics.h>

#include <windows.h>

#include <conio.h>

#include <math.h>

#include <stdio.h>

#define GRP(x, y) GetRValue(getpixel(x, y))

#define GGP(x, y) GetGValue(getpixel(x, y))

#define GBP(x, y) GetBValue(getpixel(x, y))

#define WIDTH 200 //屏幕宽

#define HEIGHT 200 //屏幕高

int main(void)
{
void CopyLayer(); //图层复制
void CarveLR(); //雕刻：自左向右，需要复制图层
initgraph(WIDTH, HEIGHT); //初始化600×600的绘图窗口
setbkcolor(WHITE); //设置背景色
cleardevice(); //清空屏幕
loadimage(NULL, “D:\lena.jpg”);
CopyLayer();
CarveLR();
getch();
closegraph();

return 0;
}

//雕刻，自左向右，本点 = 本点 - 右点 +127
void CarveLR()
{
int r, g, b;
for(int x = 0; x < WIDTH; x++)
for(int y = 0; y < HEIGHT; y++ )
{
r = GRP(x, y) - GRP(x + 1, y) + 127;
g = GGP(x, y) - GGP(x + 1, y) + 127;
b = GBP(x, y) - GBP(x + 1, y) + 127; //127为调整后的亮度，可调
putpixel(x, y, RGB(r, g, b));
}
}
//图层复制
void CopyLayer()
{
Resize(NULL, WIDTH + WIDTH, HEIGHT);
for(int x = 0; x < WIDTH; x++)
for(int y = 0; y < HEIGHT; y++ )
{
putpixel(x + WIDTH, y, getpixel(x, y));
}
}
